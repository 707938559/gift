create table user(
uid int auto_increment key,
id varchar(20) not null unique,
password char(70) not null,
nickname text not null,
sex int not null,
headPic text,
phone text,
address text,
selfIntro longtext
);

create table events(
eid int auto_increment key,
type int not null,
rid int not null,
id varchar(20) not null,
content longtext,
phone text,
address text,
createTime text
);


create table goods(
gid int auto_increment key,
eid int not null,
id varchar(20) not null
);

create table clicks(
clid int auto_increment key,
eid int not null
);


create table chats(
cid int auto_increment key,
id text not null,
rid text not null,
content text not null,
createTime text
);


create table rooms(
rid int auto_increment key,
type int not null,
roomname text not null
);

create table members(
mid int auto_increment key,
id varchar(20) not null,
rid int not null
);

create table pics(
pid int auto_increment key,
eid int not null,
url text not null
);
